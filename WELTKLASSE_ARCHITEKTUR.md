SELECT 
  v.*,
  (v.views_count * 0.3 + v.likes_count * 0.5 + v.shares_count * 0.2) AS trend_score
FROM videos v
WHERE v.created_at > NOW() - INTERVAL '24 hours'
ORDER BY trend_score DESC
LIMIT 100;
